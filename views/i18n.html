Internationalization

i18n made easy

Translating the directive's helper buttons is as easy as including one of the available i18n files from the i18n folder or create a custom texts at runtime: 

			
var angular_multi_select = angular.module('angular-multi-select');
angular_multi_select.config(function (angularMultiSelectI18nProvider) {
	angularMultiSelectI18nProvider.createTranslation('yolang', {
		CHECK_ALL: "Check'em all!",
		CHECK_NONE: 'Uncheck all that stuff!',
		RESET: 'Nah... undo!',
		SEARCH: 'Type here something...',
		CLEAR: ''
	});

	angularMultiSelectI18nProvider.setLang('yolang');
});
