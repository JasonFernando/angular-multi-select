Demo > Leafs/nodes styling

You can style the leafs/nodes with constants and/or with the data from each item. Make sure to check the node-label and the leaf-label configuration options for details.

Nodes

Define your directive:

								
<div
	angular-multi-select
	input-model="input_data"
	output-model="output_data_1"
	output-type="objects"

	dropdown-label="Click me!"
	node-label="<[ text ]> &nbsp;<b>(checked
		<# angularMultiSelectConstants.INTERNAL_KEY_CHECKED_CHILDREN #> /
		<# angularMultiSelectConstants.INTERNAL_KEY_CHILDREN_LEAFS #>)</b>"
	leaf-label="<[ text ]>"
></div>

Define your data models:

								
$scope.input_data = [
	{ text: 'A', value: 'a', id: 1},
	{ text: 'B', value: 'b', id: 2, open: true, children: [
		{ text: 'C', value: 'c', id: 3, checked: true}
	]},
	{ text: 'D', value: 'd', id: 4, children: [
		{ text: 'E', value: 'e', id: 5},
		{ text: 'F', value: 'f', id: 6, checked: true}
	]},
	{ text: 'G', value: 'g', id: 7, children: [
		{ text: 'H', value: 'h', id: 8},
		{ text: 'I', value: 'i', id: 9, children: [
			{ text: 'J', value: 'j', id: 10},
			{ text: 'K', value: 'k', id: 11}
		]},
		{ text: 'L', value: 'l', id: 12},
		{ text: 'M', value: 'm', id: 13}
	]}
];

$scope.output_data_1 = [];
